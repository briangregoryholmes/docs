## Description

Svelvet is the wrapping parent component that renders out nodes and edges. It is configurable via props, but no props are required. Many props can be set globally at this level and the overriden at the Node, Edge or Anchor level.

```HTML App.svelte
<script>
  import { Node, Svelvet, Minimap, Controls } from 'svelvet';
</script>

<Svelvet id="my-graph" width="{500}" height="{500}" TD minimap locked>
  <Node />
  <Node id="alpha" bgColor="red" label="Default Node" />
  <Controls />
</Svelvet>
```

## Props

<ResponseField name="width" type="number" default="size of wrapping element">
  Width of the graph window on the DOM. If not passed, element will fill its
  wrapping container
</ResponseField>
<ResponseField name="height" type="number" default="size of wrapping element">
  Height of the graph window on the DOM. If not passed, element will fill its
  wrapping container
</ResponseField>
<ResponseField name="id" type="number | string" default="incrementing integer">
  Identification for the graph. If not passed, defaults to an incrementing
  integer. Used as the HTML id value for the element taking the form G-id
</ResponseField>
<ResponseField name="minimap" type="boolean" default="false">
  Boolean used to render the default Minimap component. Just pass "minimap"
  rather than minimap={true}
</ResponseField>
<ResponseField name="controls" type="boolean" default="false">
  Boolean used to render the default Controls component. Just pass "controls"
  rather than controls={true}
</ResponseField>
<ResponseField name="edge" type="Edge Component">
  A custom Edge component to be used as the global Edge style for the graph. Can
  be overriden at the Node and Anchor level
</ResponseField>
<ResponseField
  name="edgeStyle"
  type="bezier | step | straight"
  default="bezier"
>
  Enum representing the global edge style for the graph. Can be overriden at the
  Node and Anchor level
</ResponseField>
<ResponseField name="snapTo" type="number" default="1">
  Pixel value at default scale that Nodes should snap to when being moved.
</ResponseField>
<ResponseField name="editable" type="boolean" default="false">
  Allows the Node properties to be edited via CTRL+click.
</ResponseField>
<ResponseField name="locked" type="boolean" default="false">
  Locks node movement, but still allows the graph to be panned
</ResponseField>
<ResponseField name="initialZoom" type="number" default="1">
  Sets the initial zoom level for the graph. Below 1 zooms out, above 1 zooms
  in.
</ResponseField>
<ResponseField
  name="theme"
  type="light | dark | purple | parchment"
  default="light"
>
  Enum controlling default theme for the graph.
</ResponseField>
<ResponseField name="mermaid" type="string" default="">
  Graph representation using Mermaid-inspired string syntax. Great for quickly
  developing simple Nodes and Edges.
</ResponseField>
<ResponseField name="TD" type="boolean" default="false">
  Boolean controlling the directionality of the graph. TD being set false by
  default means that the graph direction is inherently left-to-right. Pass "TD"
  as a prop to change the direction to top-down.
</ResponseField>
